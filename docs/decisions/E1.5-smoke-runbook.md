## 4. Post-execute spot-checks

```bash
docker compose exec trino trino
```

```sql
-- 4.1. dim_team canonical_id переключился на silver-canonical
SELECT team_id, team_name, season
FROM iceberg.gold.dim_team
WHERE team_name LIKE '%Manchester%' AND season = 2024
ORDER BY team_name;
-- EXPECT: team_id = 'manchester_united' (НЕ 'manchester_utd')
-- EXPECT: team_id = 'manchester_city'

-- 4.2. dim_player canonical_id с 'fb_' префиксом
SELECT player_id, player_name FROM iceberg.gold.dim_player LIMIT 5;
-- EXPECT: каждый player_id начинается на 'fb_'

-- 4.3. fct_player_match.player_id consistency с dim_player
SELECT COUNT(*) FROM iceberg.gold.fct_player_match
WHERE NOT regexp_like(player_id, '^fb_.+');
-- EXPECT: 0

-- 4.4. dim_match home/away_team_id матчат dim_team.team_id
SELECT COUNT(*) FROM iceberg.gold.dim_match dm
LEFT JOIN iceberg.gold.dim_team dt
  ON dm.home_team_id = dt.team_id
 AND dm.league = dt.league
 AND dm.season = dt.season
WHERE dt.team_id IS NULL AND dm.home_team_id IS NOT NULL;
-- EXPECT: 0 (или ≤ baseline orphan rate; current orphan policy: NULL home/away_team_id допустим)

-- 4.5. match_outcomes ⊆ dim_team
SELECT COUNT(*) FROM iceberg.gold.match_outcomes mo
LEFT JOIN iceberg.gold.dim_team dt
  ON mo.home_team_id = dt.team_id AND mo.league = dt.league AND mo.season = dt.season
WHERE dt.team_id IS NULL AND mo.home_team_id IS NOT NULL;
-- EXPECT: 0

-- 4.6. fct_event.team_id_canonical (E3) совпадает с dim_team.team_id (post-cutover)
WITH e3_teams AS (
  SELECT DISTINCT team_id_canonical, league,
         CASE WHEN length(season)=4 AND TRY_CAST(season AS bigint) BETWEEN 2000 AND 2100
              THEN TRY_CAST(season AS bigint)
              ELSE 2000 + TRY_CAST(substr(season,1,2) AS bigint)
         END AS season
  FROM iceberg.gold.fct_event
  WHERE team_id_canonical IS NOT NULL
)
SELECT COUNT(*) AS orphan_team_ids
FROM e3_teams e3
LEFT JOIN iceberg.gold.dim_team dt
  ON dt.team_id = e3.team_id_canonical
 AND dt.league  = e3.league
 AND dt.season  = e3.season
WHERE dt.team_id IS NULL;
-- EXPECT: значительное снижение vs pre-cutover (drift между manchester_united/manchester_utd снят).
```
